Word Count in Spark

val inputRDD = sc.textFile("/data/shakespeare.txt.gz")
val mapRDD = inputRDD.flatMap(l => l.split("\\P{L}+"))
val map2RDD = mapRDD.map(word => (word, 1))
val wordCount = map2RDD.reduceByKey((a, b) => a + b)
wordCount.collect
